性质

  1. 区分大小写
  2. 结尾不需要分号
<script>
    alert("hello world")//输出语句

    window.alert("Hello world")//弹出警告框

    document.write("Hello JS")//写入html中

    console.log("hello js")//控制台输出(F12/console)
</script>

变量

  1. JavaScript中用 Var 来声明变量
  2. JavaScript是一门弱类型语言,变量可以存放不同类型的值
<script>
    //var 定义变量
    var a = 3;
    a = "维尼";

    alert(a);//输出张三

    特点: 作用域大 全局变量 可重复定义
    {
        var x = 1;
        var x = "A"
    }
    alert(x); //输出A
</script>

  1. let 关键字相当于定义局部变量

  2. const 关键字相当于常量

    <script>
        {
            let x = 1;
        }
        alert(x); //不输出

        const pi = 3.14
        pi = 3.15
        alert(pi) //不输出
    </script>

    数据类型

  3. === (三等) 不会自动转换类型, == 则会

    <script>
        var age1 = 20;
        var age2 = "20";
        var age3 = 20;

        alert(age1 == age2); //true

        alert(age1 == age2); //false

        alert(age3 === age3); //true
    </script>

  4. 其他类型转数字

    <script>
        alert(parseInt("12")); //12
        alert(parseInt("12A45")); //12
        alert(parseInt("A12")); //NaN
    </script>

函数

  1. 形参不需要定义类型
    <script>
        //定义方式一
        funtion add(a,b) {
            return a+b;
        }

        var res = add(1,2)
        alert(res);

        //定义方式二
        var add = funtion(a,b){
            return a+b;
        }

        var res = add(1,2)
        alert(res);

        //传参个数可以是任意个,按照读入顺序满足形参个数
    </script>

Array

  1. 定义
    <script>
        //定义方式一
        var arr = new Array(1,2,3,4);

        //定义方式二
        var arr = [1,2,3,4];

        //特点: 长度可变 类型可变
        var arr = [1,2,3,4]
        arr[10] = 50;
        arr[9] = "A"
        arr[8] = true;

        console.log(arr);//[1,2,3,4,udf,udf,udf,true,A,50]
    </script>
  2. 方法
    <script>
        var arr = [1,2,3,4,5];

        //forEach: 遍历有值的数组元素,并调用一次函数
            arr.forEach(funtion(e){
                console.log(e)
            })
        //ES6: 箭头函数: (...) => {...}
            arr.forEach((e) => {
                console.log(e)
            })

        //push: 新元素添加到数组末尾,并返回新的长度
            arr.push(6,7,8);
            console.log(arr);// 1,2,3,4,5,6,7,8

        //splice: 删除元素
            arr.splice(startPosition,endPosition)
    </script>

    String

  3. 定义
    <script>
        //创建字符串对象
        var str = new String("...")
        var str = "..."
    </script>
  4. 方法
    <script>
        var str = "abc"

        //length,返回字符串长度
            str.length(); //3

        //charAt(),返回指定位置字符
            str.chatAt(0);//a
        
        //indexOf(),检索字符串
            str.indexOf("bc");//true
        
        //trim(),去除字符串两端空格
            str.trim();

        //substring,提取两坐标之间的字符
            str.substring(0,1);//ab
    </script>
