			public void actionPerformed(ActionEvent e) {
				num2 = Integer.parseInt(txt.getText());//存第二个数
				switch(op) {//计算
					case '+':
						result = num1 + num2;
						break;
					case '-':
						result = num1 - num2;
						break;
					case '*':
						result = num1 * num2;
						break;
					case '/':
						if(num2 == 0) {
							blnResult = false;
							txt.setText("除数不能为0");
						}
						else {
							result = num1 / num2;
						}
						break;
				}
				txt.setText("" + result);//显示结果(空字符串+result即可转换为String类型)
				//恢复现场
				blnNum = true;
				blnResult = true;
			}
